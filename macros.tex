% Math font
\newfam\bbfam
\font\tenbb=msbm10
\textfont\bbfam=\tenbb
\def\bb{\fam\bbfam\tenbb}

% Math symbols
\def\Z{{\bb Z}} % Integers
\def\Q{{\bb Q}} % Rationals
\def\R{{\bb R}} % Reals
\def\C{{\bb C}} % Complex numbers

% Tables
\def\crl{\cr\noalign{\hrule}}
\def\frac#1#2{{\strut\displaystyle #1\over\strut\displaystyle #2}}
\def\up#1#2{\vbox{\hbox{\rm#2}\kern #1pt}}
\def\low#1#2{\vbox{\hbox{\rm#2}\kern-#1pt}}
\def\multirow#1#2{\multispan#1\hfil#2\hfil\vrule}

% From: Levy, Seroul, "A beginner's book of TeX", p.62.
\def\iitem{\itemitem}
\def\iiitem{\par\indent\indent\hangindent3\parindent\textindent}
\def\iiiitem{\par\indent\indent\indent\hangindent4\parindent\textindent}

% Emphasis
\def\emph#1{{\bf #1}}
