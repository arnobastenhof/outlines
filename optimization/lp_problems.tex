\beginsection The Linear Programming Problem

\item{A.}\emph{Example}: managing a production facility\smallskip
\iitem{1.}Snapshot / situation at any one moment in time\smallskip
\iiitem{(a)}\emph{Outputs}: \emph{products} $j=1,\ldots,n$ with prevailing
market price $\sigma_j$ ($\$$/unit) \smallskip
\iiitem{(b)}\emph{Inputs}: $b_i$ available units of \emph{raw materials} $i=1,
\ldots,m$ w. unit market value $\rho_i$ ($\$$/unit)\smallskip
\iiitem{(c)}Units of material $i$ required for producing product $j$: $a_{ij}$
\smallskip
\iitem{2.}\emph{Assumption}: the production facility is small relative to the
market and cannot influence $\rho_i$ or $\sigma_j$\smallskip
\iitem{3.}\emph{Production manager} as optimist (\emph{resource allocation
problem})\smallskip
\iiitem{(b)}\emph{Net revenue} $c_j$ associated w. producing 1 unit of $j$:
$\sigma_j-\sum_{i=1}^m\rho_ia_{ij}$\smallskip
\iiitem{(c)}\emph{Goal}: determine \emph{production quantities} $x_j$ to
maximize the \emph{total net revenue} $\sum_{j=1}^nc_jx_j$ subject to
$x_j\geq 0$ ($j=1,\ldots,n$) and $\sum_{j=1}^na_{ij}x_j\leq b_i$ ($i=1,\ldots,
m$)\smallskip
\iitem{4.}\emph{Comptroller} as pessimist (\emph{determine cost of inventory})
\smallskip
\iiitem{(a)}\emph{Goal}: assign values $w_i$ to minimize the \emph{lost
opportunity cost} $\sum_{i=1}^mb_iw_i$ subject to $w_i\geq\rho_i$ for $i=1,
\ldots,m$ (cannot affect $\rho_i$) and $\sum_{i=1}^ma_{ij}w_i\geq\sigma_j$ for
$j=1,\ldots,n$ (cannot affect $\sigma_j$)\smallskip
\iiitem{(b)}\emph{Rephrased}, letting $y_i=w_i-\rho_i$: to minimize 
$\sum_{i=1}^mb_iy_i$ (dropping the constant $\sum_{i=1}^mb_i\rho_i$) s.t.
$y_i\geq 0$ ($1\leq i\leq m$) and $\sum_{i=1}^ma_{ij}y_i\geq c_j$ ($1\leq j\leq
n$)\smallskip

\item{B.}Formulation of \emph{linear programming problems}\smallskip
\iitem{1.}\emph{Standard form} (of \emph{primal problem}): maximize
the \emph{objective function value} $u=\sum_{j=1}^nc_{j}x_j$ subject to
\emph{constraints} $\sum_{j=1}^na_{ij}x_j\leq b_i$
($1\leq i\leq m$) and $x_j\geq 0$ ($1\leq j\leq n$)\smallskip
\iitem{2.}\emph{General form}\smallskip
\iiitem{(a)}${\rm min}\ u=-{\rm max}\ -u$\smallskip
\iiitem{(b)}$\sum_{j=1}^na_{ij}x_j\geq b_i$ iff $\sum_{j=1}^n(-a_{ij})x_j\leq
-b_i$\smallskip
\iiitem{(c)}$\sum_{j=1}^na_{ij}x_j=b_i$ iff $\sum_{j=1}^na_{ij}x_j\leq b_i$ and
$\sum_{j=1}^na_{ij}x_j\geq b_i$\smallskip
\iiitem{(d)}$\sum_{j=1}^na_{ij}x_j\leq b_i$ iff $\sum_{j=1}^na_{ij}x_j+w_i=b_i$
and $w_i\geq 0$ (for fresh \emph{slack variable} $w_i$)\smallskip
\iitem{3.}\emph{Dual problem}: min $v=\sum_{i=1}^mb_iy_i$
s.t. $\sum_{i=1}^ma_{ij}y_i\geq c_j$, $y_i\geq 0$ ($1\leq j\leq n$, $1\leq i
\leq m$)\smallskip
\iiitem{(a)}Dual of the dual is the primal\smallskip
\iiiitem{I.}Dual in primal standard form: $-{\rm max}\ \sum_{i=1}^m(-b_i)y_i$
s.t. $\sum_{i=1}^m(-a_{ij})y_i\leq(-c_j)$, $y_i\geq 0$\smallskip
\iiiitem{II.}Dual of the dual: $-{\rm min}\sum_{j=1}^n(-c_j)x_j$ s.t.
$\sum_{j=1}^n(-a_{ij})x_j\geq(-b_i)$, $x_j\geq 0$\smallskip

\item{C.}\emph{Solutions}: \'any choice of \emph{decision variables} $x_j$
($y_i$)\smallskip
\iitem{1.}\emph{Feasible solutions}: any solution satisfying all constraints
\smallskip
\iitem{2.}\emph{Optimal solutions}: any feasible solution that attains the
desired max (min)\smallskip
\iitem{3.}\emph{Infeasible problems}: any problem having no feasible solution
\smallskip
\iiitem{(a)}E.g., max $5x_1+4x_2$ s.t. $x_1+x_2\leq 2$, $-2x_1-2x_2\leq-9$,
$x_1,x_2\geq 0$\smallskip
\iitem{4.}\emph{Unbounded problems}: problems having feasible solutions with
arb. large (small) objective values\smallskip
\iiitem{(a)}E.g., max $x_1-4x_2$ s.t. $-2x_1+x_2\leq -1$, $-x_1-2x_2\leq-2$,
$x_1,x_2\geq 0$\smallskip

\item{D.}\emph{Weak Duality Theorem}: $\forall$ primal-, resp. dual feasible
$(x_1,\ldots,x_n)$, $(y_1,\ldots,y_m)$, we have $u\leq v$\smallskip
\iitem{1.}\emph{Proof}: $\sum_{j=1}^nc_jx_j\leq\sum_{j=1}^n\left(\sum_{i=1}^m
a_{ij}y_i\right)x_j=\sum_{i=1}^m\left(\sum_{j=1}^na_{ij}x_j\right)y_i\leq
\sum_{i=1}^mb_iy_i$\smallskip
\iitem{2.}\emph{Corollaries}\smallskip
\iiitem{(a)}Any primal-, resp. dual feasible $(x_1,\ldots,x_n)$, $(y_1,\ldots,
y_m)$ s.t. $u=v$ are optimal\smallskip
\iiitem{(b)}If the primal (dual) is unbounded, then the dual (primal) is
infeasible\smallskip
\iiiitem{I.}Counterexample to $\Leftarrow$: infeasibility of both the primal
`max $2x_1-x_2$ s.t. $x_1-x_2\leq 1$, $-x_1+x_2\leq -2$, $x_1,x_2\geq 0$' and
the dual `min $y_1-2y_2$ s.t. $y_1-y_2\geq 2$, $-y_1+y_2\geq -1$, $y_1,y_2\geq
0$'\smallskip

\vfill\eject
