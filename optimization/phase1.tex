\beginsection Phase I Algorithms for Infeasible Start Tableaux

\item{A.}Application of the simplex method when the initial tableau is primal-
and/or dual infeasible\smallskip
\iitem{1.}If primal- \'and dual feasible: initial tableau is optimal\smallskip
\iitem{2.}If primal feasible and dual infeasible: apply primal simplex method
\smallskip
\iitem{3.}If primal infeasible and dual feasible: apply dual simplex method
\smallskip
\iitem{4.}If primal- \'and dual infeasible: apply one of methods B, C, D
\smallskip
\item{B.}\emph{Primal algorithm}\smallskip
\iitem{1.}\emph{Auxiliary problem}:
max $\xi=-x_0$ s.t. $\sum_{j=1}^na_{ij}x_j-x_0\leq b_i$ ($1\leq i\leq m$),
$x_0,x_j\geq 0$ ($1\leq j\leq n$)\smallskip
\iiitem{(a)}Auxiliary problem is primal feasible: $x_j\leftarrow 0$, $x_0
\leftarrow\max_{1\leq i\leq m}|b_i|$\smallskip
\iiitem{(b)}Original problem is primal feasible iff auxiliary problem has
optimal solution $\xi=0$ (so $x_0=0$)\smallskip
\iitem{2.}Initialize tableau, $P$ now a permutation over $\{0,\ldots,n+m\}$
\smallskip

\iitem{3.}Pivot on $a_{lk}$ for $k=0$ and $l=\min_{1\leq i\leq m}b_i(<0)$
\smallskip
\iitem{4.}Assert resulting tableau is primal feasible and apply primal simplex
method to maximize $\xi$\smallskip
\iitem{5.}If $\xi\not=0$, halt (problem is primal- and dual infeasible). Else,
tableau is primal feasible.
\smallskip
\iitem{6.}Apply primal simplex method to maximize $u$, ignoring col $P^{-1}(0)$
($x_0$)\smallskip
\smallskip

\item{C.}\emph{Dual primal algorithm}\smallskip
\iitem{1.}Introduce the \emph{modified primal objective} $\eta=\sum_{j=1}^n-x_j$,
so the modified problem is dual feasible\smallskip
\iitem{3.}Apply dual simplex method to maximize $\eta$\smallskip
\iitem{4.}Half if modified problem is dual unbounded (original problem is
primal infeasible)\smallskip
\iitem{5.}Apply primal simplex method to maximize $u$\smallskip

\item{D.}\emph{Primal dual algorithm}\smallskip
\iitem{1.}Introduce \emph{modified ineq constraints} $\sum_{j=1}^na_{ij}
x_j\leq 1$, so the modified problem is primal feasible\smallskip
\iitem{2.}Apply primal simplex method to maximize $u$\smallskip
\iitem{3.}Halt if modified problem is primal unbounded (original problem is
dual infeasible)\smallskip
\iitem{4.}Reintroduce $\bar{b}_i$ and apply dual simplex method to maximize $u$
\smallskip
